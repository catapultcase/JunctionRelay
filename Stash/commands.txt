// PULL LATEST IMAGE
docker pull rusty6285/junctionrelayserver:latest

// START IMAGE WITHOUT PASSEDTHROUGH USB
docker run -p 5000:5000 -v /dev:/dev rusty6285/junctionrelayserver:latest

// START IMAGE WITH PASSEDTHROUGH USB
